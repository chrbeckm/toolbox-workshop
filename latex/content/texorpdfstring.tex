\subsection{\titlecmd{texorpdfstring}}

\begin{frame}[fragile]{\titlecmd{texorpdfstring}}
  \begin{block}{Motivation}
    \begin{itemize}
      \item \texttt{hyperref} bezieht den Text für seine Bookmarks, z.B. aus \mintinline{latex}+\section{ }+
      \item[→] Bookmarks mögen meist keine \LaTeX-Befehle
      \item \texttt{hyperref} fängt vieles ab, aber nicht alles
      \item
        \mintinline{latex}!\section{$\alpha + 1$}! ergibt Fehlermeldung:
        \begin{minted}{latex}
          Package hyperref Warning:
          Token not allowed in a PDFDocEncoded string:
          ! Improper alphabetic constant.
        \end{minted}
    \end{itemize}
  \end{block}
  \begin{block}{Lösung}
    \begin{minted}{latex}
      \texorpdfstring{LaTeX-Code}{Unicode-Äquivalent}
    \end{minted}
  \end{block}
  \begin{block}{Beipiel}
    \begin{minted}{latex}
      \section{\texorpdfstring{$\alpha + 1$}{α + 1}}
    \end{minted}
  \end{block}
\end{frame}
