\begin{frame}[fragile]{Errors}
  \begin{itemize}
    \item Alles kaputt. Was nun?
    \item Fehlermeldungen anfangs (und teils auch später) etwas kryptisch.
  \end{itemize}
  \begin{block}{Code}
    \begin{minted}{latex}
      Ich begrüße euch mit einem \enqote{Hallo Welt}
    \end{minted}
  \end{block}
  \begin{figure}
    \centering
    \begin{minted}{text}
      (/usr/local/texlive/2015/texmf-dist/tex/latex/latexconfig/epstopdf-sys.cfg))
      ! Undefined control sequence.
      l.8 Ich begrüße euch mit einem \enqote
                                       {Hallo Welt}!
      462 words of node memory still in use:
    \end{minted}
  \end{figure}
  \huge
  \begin{itemize}
    \item<2->[$\Rightarrow$] Vertippt (es fehlt ein \verb+u+ in \mintinline{latex}+\enquote+)
  \end{itemize}
\end{frame}
